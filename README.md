```

```

## Usage

```javascript
import {Stack} from "@jeremiah_tenbrink/stack";

const stack = new Stack();
stack.push("some item")
stack.push("some item2")
stack.push("some item3")

while (!stack.isEmpty()){
  const current = stack.pop();
  console.log(current)
}
```
